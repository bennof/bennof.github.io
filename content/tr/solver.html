// Javascript // # Analytic solution (for comparison) var f = new TR.math.Func("real solution", (t) => {return 0.5*(Math.exp(2*t-3)-Math.exp(-3));}) // # Euler method approximation // setup Euler solver var euler = new TR.math.solver.EulerSolver( [function(t,x) {return Math.exp(2*t-3)}]); // run approx. and extract first column as array var euler_data = TR.data.array2d_col(euler.run([0],5,1),0); // # Heun method approximation (same result as euler) // setup Heun solver var heun = new TR.math.solver.HeunSolver( [function(t,x) {return Math.exp(2*t-3)}]); // run approx. and extract first column as array var heun_data = TR.data.array2d_col(euler.run([0],5,1),0); // # Runge Kutta method approximation (same result as analytic) // setup Runge Kutta solver var rk = new TR.math.solver.RungeKuttaSolver( [function(t,x) {return Math.exp(2*t-3)}]); // run approx. and extract first column as array var rk_data = TR.data.array2d_col(rk.run([0],5,1),0); // # Plot var plot = new TR.math.plot.Plot("Comparison initial value problem"); plot.x(TR.math.seq(0,5)); // set x axis from 0 to 15 plot.line(f.seq(5),f.toString(),"#af5431"); plot.line(euler_data,"Euler Method","#33658A"); plot.line(heun_data,"Heun Method","#5DA399"); plot.line(rk_data,"Runge Kutta Method","#f1cf0b"); plot.draw("tr_solver_example"); //draw
